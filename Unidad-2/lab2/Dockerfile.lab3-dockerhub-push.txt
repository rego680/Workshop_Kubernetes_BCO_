###############################################
# LAB 3 — UNIDAD 2: DockerHub — Cuenta y Push
# Archivo: Dockerfile.lab3-dockerhub-push
#
# Objetivo: Practicar el flujo completo de
#   crear cuenta → login → build → tag → push
#
# Pasos:
#   1. Crear cuenta en https://hub.docker.com
#   2. docker login -u TU_DOCKER_ID
#   3. docker build -t flask-hub:1.0.0 -f Dockerfile.lab3-dockerhub-push .
#   4. docker tag flask-hub:1.0.0 TU_DOCKER_ID/flask-hub:1.0.0
#   5. docker push TU_DOCKER_ID/flask-hub:1.0.0
#   6. Verificar en https://hub.docker.com/r/TU_DOCKER_ID/flask-hub
#
# Run local:
#   docker run -d --name flask-hub -p 5000:5000 flask-hub:1.0.0
# Test:
#   curl http://localhost:5000
###############################################

FROM python:3.12-slim

LABEL maintainer="DevOps Lab - Unidad 2"
LABEL description="Lab 3 - Imagen para practicar push a DockerHub"
LABEL lab="lab3-dockerhub-push"
LABEL version="1.0.0"

# Directorio de trabajo
WORKDIR /app

# Dependencias (cache)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Código fuente
COPY app.py .

# Seguridad: usuario no-root
RUN adduser --disabled-password --gecos "" appuser \
    && chown -R appuser:appuser /app
USER appuser

# Puerto
EXPOSE 5000

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:5000/health')" || exit 1

# Ejecutar
CMD ["python", "app.py"]
